# HIVE — Python dependencies
# Apple Silicon (M-series) setup — uses MPS backend, not CUDA.
#
# Recommended setup:
#   conda create -n hive python=3.10
#   conda activate hive
#
# Step 1 — install torch (standard pip, MPS support is built in for macOS):
#   pip install torch torchvision
#
# Step 2 — install PyTorch Geometric:
#   pip install torch-geometric
#   pip install torch-scatter torch-sparse -f https://data.pyg.org/whl/torch-2.x.0+cpu.html
#   (replace 2.x.0 with your installed torch version, e.g. 2.3.0)
#
# Step 3 — install remaining deps:
#   pip install -r requirements_claude.txt

# --- GUI ---
PyQt5
